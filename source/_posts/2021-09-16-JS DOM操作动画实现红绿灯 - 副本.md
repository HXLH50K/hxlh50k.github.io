重点在于`function go()`的异步递归调用。

```html
<!DOCTYPE html>

<head>
  <meta charset="utf-8" />
  <title>红绿灯</title>
  <style>
    #outer {
      width: 100px;
      height: 300px;
      border: 1px solid black;
      display: flex;
      justify-content: center;
      align-items: center;
      flex-direction: column;
    }
    .light {
      width: 80px;
      height: 80px;
      border: 1px solid black;
      border-radius: 50%;
      margin: auto;
    }
  </style>
</head>

<body>
  <div id="outer">
    <div class="light" id="red"></div>
    <div class="light" id="yellow"></div>
    <div class="light" id="green"></div>
  </div>
  <script>
    window.onload = function () {
      function changeColor(color) {
        if (color == 'red') {
          r.style.background = color
          y.style.background = 'white'
          g.style.background = 'white'
        }
        if (color == 'yellow') {
          r.style.background = 'white'
          y.style.background = color
          g.style.background = 'white'
        }
        if (color == 'green') {
          r.style.background = 'white'
          y.style.background = 'white'
          g.style.background = color
        }
      }
      function setColor(color, time) {
        return new Promise((resolve, reject) => {
          changeColor(color)
          setTimeout(resolve, time)
        })
      }
      let r, y, g
      ;[r, y, g] = document.getElementsByClassName('light')
      async function go() {
        await setColor('red', 3000)
        await setColor('yellow', 2000)
        await setColor('green', 1000)
        go()
      }
      go()
    }
  </script>
</body>
```
